const multer = require('multer');
const path = require('path');

// Configuration de stockage pour multer
const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    cb(null, 'uploads/'); // Dossier où les fichiers sont sauvegardés
  },
  filename: (req, file, cb) => {
    cb(null, `${Date.now()}-${file.originalname}`);
  }
});

// Configuration de Multer avec vérification du champ
const upload = multer({
  storage: storage,
  fileFilter: (req, file, cb) => {
    // Vérifie que le champ du fichier est bien 'fichiers'
    if (file.fieldname !== 'fichiers') {
      return cb(new multer.MulterError('LIMIT_UNEXPECTED_FILE'), false);
    }
    cb(null, true);
  }
});

module.exports = upload;
