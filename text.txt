import React, { useState } from 'react';
import Conversations from './Conversations';  // Assure-toi que l'import est correct
import ChatWindow from './ChatWindow';
import ChatDetails from './ChatDetails';

interface User {
  id: number;
  username: string;
  photo?: string;   
}

interface Discussion {
  id: number;
  photo: string;
  username: string;
}

interface Etudiant {
  id: number;
  photo: string;
  username: string;
  role: string;
}

const Messages: React.FC = () => {
  const [selectedUser, setSelectedUser] = useState<User | null>(null);

  const handleSelectUser = (user: Discussion | Etudiant) => {
    // Transformer un Discussion ou Etudiant en User
    const transformedUser: User = {
      id: user.id,
      username: user.username,
      photo: 'photo' in user ? user.photo : undefined,  // Vérifie si l'utilisateur a une photo
    };
    setSelectedUser(transformedUser);  // Définit l'utilisateur sélectionné comme un User
  };

  return (
    <div className="h-screen flex flex-col">
      {/* Layout principal pour les messages */}
      <div className="flex flex-grow bg-gray-100 overflow-hidden">
        {/* Liste des conversations */}
        <div className="flex-none w-1/3 h-full overflow-auto">
          <Conversations onSelectUser={handleSelectUser} />
        </div>
        
        {/* Fenêtre de discussion */}
        <div className="flex-grow h-full overflow-hidden">
          {selectedUser ? (
            <ChatWindow user={selectedUser} /> 
          ) : (
            <div className="flex items-center justify-center h-full">
              <p>Sélectionnez un utilisateur pour commencer la discussion</p>
            </div>
          )}
        </div>
        
        {/* Détails de la discussion */}
        <div className="flex-none w-1/3 h-full overflow-auto">
          <ChatDetails />
        </div>
      </div>
    </div>
  );
};

export default Messages;
