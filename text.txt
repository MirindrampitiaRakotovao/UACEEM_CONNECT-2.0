import React, { useState } from 'react';
import EditProfileModal from './EditProfileModal'; // Assurez-vous que le chemin est correct

const ProfilPageAdmin: React.FC = () => {
  const [isOpen, setIsOpen] = useState(false);

  // Fonction pour ouvrir le modal
  const openModal = () => {
    setIsOpen(true);
  };

  // Fonction pour fermer le modal
  const closeModal = () => {
    setIsOpen(false);
  };

  return (
    <div className="profil-page-admin">
      <h1 className="text-2xl font-bold">Profile Admin Page</h1>
      
      {/* Edit Profile Button */}
      <div className="mt-4 text-center">
        <button 
          className="bg-transparent border-blue-500 border-2 text-black py-2 px-4 rounded-lg w-full hover:bg-blue-500 hover:text-white" 
          onClick={openModal} // Utilisation correcte de la fonction openModal
        >
          Edit profile
        </button>
      </div>

      {/* Modal */}
      {isOpen && (
        <EditProfileModal 
          isOpen={isOpen} 
          closeModal={closeModal} // Passe la fonction pour fermer le modal
        />
      )}

      {/* Tabs Section */}
      <div className="tabs-section mt-8">
        {/* Autres contenus de votre page */}
      </div>
    </div>
  );
};

export default ProfilPageAdmin;
