const multer = require('multer');

// Configuration de l'emplacement de stockage des fichiers
const storage = multer.diskStorage({
  destination: function (req, file, cb) {
    cb(null, 'uploads/'); // Répertoire où seront stockés les fichiers
  },
  filename: function (req, file, cb) {
    cb(null, Date.now() + '-' + file.originalname); // Nomme le fichier avec un timestamp
  }
});

// Configuration de Multer, avec une limite de taille de fichier
const upload = multer({
  storage: storage,
  limits: { fileSize: 5 * 1024 * 1024 } // Limite de taille fixée à 5MB
});

module.exports = upload;
